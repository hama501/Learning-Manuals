typedef struct{
    float v0;
    float v3;
    float t_smooth;
    float t_straight;
    float t1;
    float t2;
    float t3;
    float delta_v;
    float changing;
}Curve;

Curve profile[4];

void Curve_Init(Curve *c, float v0, float v3, float t_smooth, float t_straight){
    c->v0 = v0;
    c->v3 = v3;
}

float Curve_GetVelocity(Curve *c, float t){
    float diff = c->v3 - c->v0;
    float accel = diff / 0.1f;
    c->v0 += accel * dt;

    if((diff > 0 && c->v0 > c->v3) ||
       (diff < 0 && c->v0 < c->v3)){
        c->v0 = c->v3;
    }

    return c->v0;
}

typedef struct{
    float p_gain;
    float i_gain;
    float integral;
    float error;
    float current_max;
}PI;

PI pi[4];

float PI_control(PI *pi, float target_vel, float now_vel){

    float error = target_vel - now_vel;

    if(integral_reset == 1){
        pi->integral = 0;
        error = 0;
    }

    pi->integral += error * dt;

    float limit = pi->current_max / pi->i_gain;
    if(pi->integral > limit) pi->integral = limit;
    if(pi->integral < -limit) pi->integral = -limit;

    float current = pi->p_gain * error + pi->i_gain * pi->integral;

    if(current > pi->current_max) current = pi->current_max;
    if(current < -pi->current_max) current = -pi->current_max;

    pi->error = error;

    return current;
}

/* -------------------- CAN受信 -------------------- */

void HAL_CAN_RxFifo0MsgPendingCallback(CAN_HandleTypeDef *hcan){

    CAN_RxHeaderTypeDef RxHeader;
    uint8_t RxData[8];

    if (HAL_CAN_GetRxMessage(hcan, CAN_RX_FIFO0, &RxHeader, RxData) != HAL_OK)
        return;

    uint32_t id = (RxHeader.IDE == CAN_ID_STD) ? RxHeader.StdId : RxHeader.ExtId;

    if(hcan == &hcan1){
        if(id >= 0x010 && id <= 0x013){
            int idx = id - 0x010;
            memcpy(&target_vel[idx], RxData, sizeof(float));
            if(target_vel[idx] != last_target_vel[idx]){
                target_flag[idx] = 1;
            }
            last_target_vel[idx] = target_vel[idx];
        }
        else if(id == 0x003){
            integral_reset = RxData[0];
            for(int i=0; i<4; i++){
				target_vel[i] = 0;
				now_vel[i] = 0;
			}
        }else if(id == 0x004){
        	if(mode_count == 0){
        		mode_count = 1;
        		control_mode = 1;
        		for(int i=0; i<4; i++){
        			integral[i] = 0;
        			pi[i].integral = 0;
        		}
        	}else if(mode_count == 1){
        		mode_count = 0;
        		control_mode = 0;
        		for(int i=0; i<4; i++){
        			integral[i] = 0;
					pi[i].integral = 0;
				}
        	}
        }
    }
    else if(hcan == &hcan2){
        if(id >= 0x201 && id <= 0x204){
            int idx = id - 0x201;
            int16_t vel_rpm = (int16_t)(RxData[2] << 8 | RxData[3]);
            now_vel[idx] = vel_rpm * 3.141592f / 30.0f / gear_ratio;
        }
    }
}

/* -------------------- CAN送信（RoboMaster電流） -------------------- */

void send_current_data(){

    uint8_t TxData[8];
    CAN_TxHeaderTypeDef TxHeader;
    uint32_t TxMailbox;

    for(int i=0;i<4;i++){
        TxData[2*i]     = current[i] >> 8;
        TxData[2*i + 1] = current[i] & 255;
    }

    TxHeader.StdId = 0x200;
    TxHeader.RTR   = CAN_RTR_DATA;
    TxHeader.IDE   = CAN_ID_STD;
    TxHeader.DLC   = 8;

    HAL_CAN_AddTxMessage(&hcan2, &TxHeader, TxData, &TxMailbox);
}

/* -------------------- CAN送信（PCへ現在速度） -------------------- */

void send_velocity_data(int motor_idx){

    CAN_TxHeaderTypeDef TxHeader;
    uint32_t TxMailbox;
    uint8_t TxData[8];

    TxHeader.StdId = 0x10F + (motor_idx << 4);
    TxHeader.RTR   = CAN_RTR_DATA;
    TxHeader.IDE   = CAN_ID_STD;
    TxHeader.DLC   = 4; // floatは4バイト

    memcpy(TxData, &now_vel, sizeof(float));

    HAL_CAN_AddTxMessage(&hcan1, &TxHeader, TxData, &TxMailbox);
}

/* -------------------- 1ms制御 -------------------- */

void HAL_TIM_PeriodElapsedCallback(TIM_HandleTypeDef *htim){

    if(htim == &htim7){
    	if(control_mode == 0){
			for(int i=0;i<4;i++){
				if(target_flag[i]){
					Curve_Init(&profile[i], now_vel[i], target_vel[i], 0, 0);
					target_flag[i] = 0;
				}
			}
			for(int i=0;i<4;i++){
				float v = Curve_GetVelocity(&profile[i], 0);
				current[i] = (int)PI_control(&pi[i], v, now_vel[i]);
			}
    	}else if(control_mode == 1){
    		for(int i = 0;i < 4;i++){
				float error = target_vel[i] - now_vel[i];
				current[i] = p_gain[i] * error + i_gain[i] * integral[i];
				if(integral_reset == 0){
					integral[i] += error * dt;
				}else if(integral_reset == 1){
					integral[i] = 0;
					error = 0;
					target_vel[i] = 0;
					now_vel[i] = 0;
				}
				if(current[i] < -10000) current[i] = -10000;
				if(current[i] > 10000) current[i] = 10000;
    		}
    	}

        send_current_data();
        send_velocity_data(count);

        if(count >= 3)
            count = 0;
        else
            count++;
    }
}
